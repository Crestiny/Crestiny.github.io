#include <iostream>
#include <string>
#include <vector>

using namespace std;

//字符串除法
string divide(string str,int x){
	int remainder=0;//高位余数 
	int current;
	for(int i=0;i<str.size();i++){
		current=remainder*10+(str[i]-'0');//求现在的被除数
		str[i]=current/x+'0';//将商保存到这一位数字
		remainder=current%x; //计算这次的余数 
	}
	
	//最后的结果，要处理一下去掉高位的0 
	int index=0;
	while(str[index]=='0'){//这里别写成0！！！！ 
		index++;
	} 
	
	return str.substr(index); 
} 

//字符串乘法 
string multiple(string str,int x){
	int carry=0;
	int current;
	for(int i=str.size()-1;i>=0;i--){//注意乘法是从最低位往前面乘的 
		current=carry+(str[i]-'0')*x;
		str[i]=current%10+'0';
		carry=current/10; 
	}
	
	//处理最高位进位 
	if(carry!=0){
		str="1"+str; 
	}
	return str;
} 

//字符串加法 
string add(string str,int x){
	int carry=x;
	int current;
	for(int i=str.size()-1;i>=0;i--){
		current=carry+(str[i]-'0');
		str[i]=current%10+'0';
		carry=current/10;
	}
	
	if(carry!=0){
		str="1"+str;
	}
	return str;
} 

int main(void){
	string str;
	while(cin>>str){
		//原数字转2进制 
		vector<int> binary;
		int last;
		while(str.size()!=0){
			last=str[str.size()-1]-'0';
			binary.push_back(last%2);
			str=divide(str,2);
		} 
		//binary里面是倒着放的二进制
		
		//二进制转十进制 
		string answer="0";
		for(int i=0;i<binary.size();i++){
			answer=multiple(answer,2);	//说实话，不是很理解先乘后加…… 
			answer=add(answer,binary[i]);
		}
		cout<<answer<<endl; 
	}
	return 0;
}

清华的题……表面看起来平平无奇，实际上难得一匹

呜呜呜我好菜啊

  • 学习

    这道题的一个要点在于，大于十进制的话，有的数字要用字符表示

    要求从M进制转为N进制，那么应该……找十进制作为中介，M—>10——>N

    • M—>10的方法：

      将每一位数字乘上这一位的权重：k^(n-1)

    • 10—>N的方法：

      循环，对N取模，除N

#include <iostream>
#include <string>
#include <vector>

using namespace std;

int char2int(char c){
	int result;
	if(c>='0'&&c<='9'){
		result=c-'0';
	}else{
		result=c-'A'+10;
	}
	return result;	
}

char int2char(int i){
	char result;
	if(i>=0&&i<=9){
		result=i+'0';
	}else{
		result=i-10+'a';
	}
	return result;
}

int main(void){
	int M,N;
	while(cin>>M>>N){
		string x;
		cin>>x;
		
		//m进制转10进制 
		long long num=0;
		for(int i=0;i<x.size();i++){//为什么感觉这里是反的？？ ——懂了，没反…… 
			num*=M;
			num+=char2int(x[i]);
		}
		
		vector<char> answer;
		while(num!=0){
			answer.push_back(int2char(num%N));
			num/=N;
		} 
		
		for(int i=answer.size()-1;i>=0;i--){
			printf("%c",answer[i]);
		}
		cout<<endl;
	}
	return 0;
}

我知道M->10的操作是什么意思了……

是每次加完高位的数字，然后乘以M，也就是把它往前推一位，升一下它的权重

所以其实没反，高位一直乘到最后，它还是拥有最高权重的那个
