#include <iostream>
#include <cstdio>
#include <string>
#include <algorithm>

using namespace std;

struct grade{
	int id;
	string p;
	int q;
};

bool compare0(struct grade g1,struct grade g2){
	if(g1.q>g2.q){
		return 1;
	}else if(g1.q==g2.q){
		if(g1.id<g2.id){
			return 1;
		}else{
			return 0;
		}
	}else if(g1.q<g2.q){
		return 0;
	}else{
		return 0;
	}
}

bool compare1(struct grade g1,struct grade g2){
	if(g1.q<g2.q){
		return 1;
	}else if(g1.q==g2.q){
		if(g1.id<g2.id){
			return 1;
		}else{
			return 0;
		}
	}else if(g1.q>g2.q){
		return 0;
	}else{
		return 0;
	}
}

int main(void){
	int n;
	int order;
	while(cin>>n>>order){
		struct grade g[n];
		for(int i=0;i<n;i++){
			cin>>g[i].p>>g[i].q;
			g[i].id=i;
		}
		
		if(order==0){
			sort(g,g+n,compare0);
		}else{
			sort(g,g+n,compare1);
		}
		
		for(int i=0;i<n;i++){
			cout<<g[i].p<<" "<<g[i].q<<endl;
		}		
		
	} 
    return 0;
}

¶整数奇偶排序

¶代码

#include <iostream>
#include <cstdio>
#include <algorithm>

using namespace std;

bool compare(int a,int b){
	return a>b;
}

int main(void){
	int a[10],odd[10],even[10];
	while(cin>>a[0]>>a[1]>>a[2]>>a[3]>>a[4]>>a[5]>>a[6]>>a[7]>>a[8]>>a[9]){
		int j=0,k=0;
		for(int i=0;i<10;i++){
			if(a[i]%2==0){
				even[j++]=a[i];
			}else{
				odd[k++]=a[i];
			}
		}
		
		sort(odd,odd+k,compare);//降序 
		sort(even,even+j);
		
		for(int i=0;i<k;i++){
			cout<<odd[i]<<" ";
		}
		
		for(int i=0;i<j;i++){
			cout<<even[i]<<" ";
		}
		cout<<endl;
	}
    return 0;
}

#include <iostream>
#include <cstdio>
#include <algorithm>

using namespace std;

struct country{
	int id;//国家号 
	int rk1;
	int rk2;
	int rk3;
	int rk4;
	int gold;
	int medal;
	int people;	
};

bool compare1(struct country c1,struct country c2){
	return c1.gold>c2.gold;
}

bool compare2(struct country c1,struct country c2){
	return c1.medal>c2.medal;
}

bool compare3(struct country c1,struct country c2){
	return (c1.gold*1.0/c1.people)>(c2.gold*1.0/c2.people);
}

bool compare4(struct country c1,struct country c2){
	return (c1.medal*1.0/c1.people)>(c2.medal*1.0/c2.people);
}

bool compareId(struct country c1,struct country c2){
	return (c1.id)<(c2.id);
}

int main(void){
	int n,m;
	while(cin>>n>>m){
		struct country c1[n];//最初输入的国家 
		struct country c2[m];//真正要排序的国家 
		for(int i=0;i<n;i++){
			c1[i].id=i;
			cin>>c1[i].gold>>c1[i].medal>>c1[i].people;
		}
		
		int sortCountry[m];
		
		for(int i=0;i<m;i++){//把需要排名的国家号放进另一个数组里 
			cin>>sortCountry[i];
		}
		
		int j=0;
		
		for(int i=0;i<m;i++){
			c2[j++]=c1[sortCountry[i]];//根据国家号去取数据，放入c2； 
		}
		
		//开始进行四种排序
		//第一种 
		sort(c2,c2+m,compare1);
		for(int i=0;i<m;i++){
			//先按顺序赋值下来，然后再改一道
			c2[i].rk1=i+1; 
			//如果有数值相同的，要赋一样的排名值
		}
		
		for(int i=0;i<m;){
			int j=1;
			while(c2[i].gold==c2[i+j].gold&&(i+j<m)){
				c2[i+j].rk1=c2[i].rk1;
				j++;
			}
			i=i+j;
		}		
		
		//第二种 
		sort(c2,c2+m,compare2);
		for(int i=0;i<m;i++){
			//先按顺序赋值下来，然后再改一道
			c2[i].rk2=i+1; 
			//如果有数值相同的，要赋一样的排名值
		}

		for(int i=0;i<m;){
			int j=1;
			while(c2[i].medal==c2[i+j].medal&&(i+j<m)){
				c2[i+j].rk2=c2[i].rk2;
				j++;
			}
			i=i+j;
		} 				
		
		//第三种 
		sort(c2,c2+m,compare3);
		for(int i=0;i<m;i++){
			//先按顺序赋值下来，然后再改一道
			c2[i].rk3=i+1; 
			//如果有数值相同的，要赋一样的排名值
		}
		
		for(int i=0;i<m;){
			int j=1;
			while((c2[i].gold*1.0/c2[i].people)==(c2[i+j].gold*1.0/c2[i+j].people)&&(i+j<m)){
				c2[i+j].rk3=c2[i].rk3;
				j++;
			}
			i=i+j;
		}		
		
		//第四种 
		sort(c2,c2+m,compare4);
		for(int i=0;i<m;i++){
			//先按顺序赋值下来，然后再改一道
			c2[i].rk4=i+1; 
			//如果有数值相同的，要赋一样的排名值
		}
		
		for(int i=0;i<m;){
			int j=1;
			while((c2[i].medal*1.0/c2[i].people)==(c2[i+j].medal*1.0/c2[i+j].people)&&(i+j<m)){
				c2[i+j].rk4=c2[i].rk4;
				j++;
			}
			i=i+j;
		}		
		
		//这里要恢复按国家号的排名！！！
		sort(c2,c2+m,compareId);	
		
		int best_rk;
		for(int i=0;i<m;i++){
			
			best_rk=min(c2[i].rk1,min(c2[i].rk2,min(c2[i].rk3,c2[i].rk4)));
			if(best_rk==c2[i].rk1){
				cout<<best_rk<<":"<<1<<endl;
				continue;
			}
			
			if(best_rk==c2[i].rk2){
				cout<<best_rk<<":"<<2<<endl;
				continue;
			}	
			
			if(best_rk==c2[i].rk3){
				cout<<best_rk<<":"<<3<<endl;
				continue;
			}
			
			if(best_rk==c2[i].rk4){
				cout<<best_rk<<":"<<4<<endl;
				continue;
			}
										
		}
		cout<<endl; 
	} 
    return 0;
}

¶总结

说下之前错在哪：

  • 首先我检查了那个“修改排名”的函数，发现不应该在for循环里面加i++
  • 确定这个函数没问题之后，我怀疑输出有问题，仔细一想：我没在最后输出的时候恢复按国家号的排名啊！呜呜呜，难怪第二行第三行的数字一直是反的

获得一些经验：

  • 不能用调试的话，从小处检查起，确保小函数是正确的
  • 写代码不要想当然，比如直接在for循环里面加i++这档子事，还是要过脑的
  • 不要隔太久写，过久了就更不知道怎么写了
