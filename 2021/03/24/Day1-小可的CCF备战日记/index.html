/* CCF201912-2 回收站选址 */

#include <bits/stdc++.h>

using namespace std;

const int N = 1000;
pair<int, int> p[N];	//p是存点的数组
const int CN = 4;
int cnt[CN + 1];	//算每个分数对应的点的数量

int main()
{
    int n;
    map<pair<int, int>, int> ps;

    scanf("%d", &n);
    for(int i = 0; i < n; i++) {
        int x, y;
        scanf("%d%d", &x, &y);

        p[i] = make_pair(x, y);
        ps[p[i]] = 1;	//使用map作为数据结构，key是点，value是
    }

    memset(cnt, 0, sizeof(cnt));	//将cnt数组元素全部置0
    for(int i = 0; i < n; i++) {
        int x = p[i].first;
        int y = p[i].second;	//原来pair元素可以这样用~

        if(ps[make_pair(x, y -1)] && ps[make_pair(x, y + 1)] &&
                ps[make_pair(x - 1, y)] && ps[make_pair(x + 1, y)])
            cnt[ps[make_pair(x - 1, y - 1)] + ps[make_pair(x - 1, y + 1)] +
                    ps[make_pair(x + 1, y - 1)] + ps[make_pair(x + 1, y + 1)]]++;	//太妙了
    }

    for(int i = 0; i <= CN; i++)
        printf("%d\n", cnt[i]);

    return 0;
}
