  • 分析：

    其实就是实现有运算符优先级的四则运算

    有点想用stack，遇到乘号就pop，push结果

    好像又有点不对……

  • 学习：

    想想还是学习一下https://www.geek-share.com/detail/2794345166.html

    先把中缀表达式改成后缀，再用逆波兰算后缀？

    遇到一个问题：怎么比较优雅地描述运算符的优先级

    在上面的帖子里我学到的是给运算符编号

    居然没有现成的清空stack的函数，只能搞个while循环了

    转成后缀以后，用逆波兰来算：https://zh.wikipedia.org/wiki/逆波兰表示法

  • 错误：

    • 记录一个神奇的错误：

    这里好像空栈不能调用top？

    • 过了样例但只有70分

      #include <iostream> 
#include <stack>

using namespace std;

//ASCII码：+是43，-是45，x是120，/是47 
//优先级大小和ASCII码大小好像…… 

int get_prior(char c){
	switch(c){
		case '+':case '-':
			return 1;
		case 'x':case '/':
			return 2;	
	}
} 

int main(void){
	int n;
	while(cin>>n){
		char reverse[n][7];
		char middle[n][7];
		stack<char> transform;
		stack<char> calc; 
		
		//先读进字符数组里面吧 
		for(int i=0;i<n;i++){
			cin>>middle[i][0]>>middle[i][1]>>middle[i][2]>>middle[i][3]>>middle[i][4]>>middle[i][5]>>middle[i][6];
		}

		//中缀转后缀 
		char temp,top_op;
		for(int i=0;i<n;i++){
			int k=0;
			//每处理一行表达式，都要清空transform栈，因为它只是处理运算符的中转站
			//但是仔细一想，好像每次处理完都会全部弹出去了？ 
			for(int j=0;j<7;j++){
				temp=middle[i][j];
				if(temp>=48&&temp<=57){	//是操作数的话，直接存到后缀表达式的字符数组里面 
					reverse[i][k++]=temp;
				}else{//如果是操作符 
					 if(transform.empty()){//栈为空可以直接压栈 
					 	transform.push(temp);
					 }else{//栈不空就比较优先级 
					 	top_op=transform.top();//获取栈顶运算符
						if(get_prior(temp)>get_prior(top_op)){
							transform.push(temp);
						}else{
							while(get_prior(top_op)>=get_prior(temp)&&(!transform.empty())){
								reverse[i][k++]=transform.top();
								transform.pop();
								if(transform.empty()){
									break;//栈空了就直接退出while 
								}else{
									top_op=transform.top();
								} 
							}
							transform.push(temp);
						}
					 }
				}

			}
			//各字符处理完毕，栈中剩余的运算符直接出栈 
			while(!transform.empty()){
				reverse[i][k++]=transform.top();
				transform.pop(); 
			} 			
		}

		//后缀运算，把每个表达式利用calc来做计算
		for(int i=0;i<n;i++){
			for(int j=0;j<7;j++){
				temp=reverse[i][j];
				if(temp>=48&&temp<=57){
					calc.push(temp);
				}else{
					int op2=calc.top()-48;
					calc.pop();
					int op1=calc.top()-48;
					calc.pop();
					switch(temp){
						case '+':
							calc.push(op1+op2+'0');break;
						case '-':
							calc.push(op1-op2+'0');break;	
						case 'x':
							calc.push(op1*op2+'0');break;	
						case '/':
							calc.push(op1/op2+'0');break;	
					}
				}
			}
			//看结果是否等于24 
			if(calc.top()-48==24){
				cout<<"Yes"<<endl;
			}else{
				cout<<"No"<<endl;
			}
		}
		 
		
	}
	return 0;
}
